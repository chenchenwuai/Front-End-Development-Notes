###### 这是一个移动端输入组件，会被键盘整体顶起的那种
index.jsx
```jsx
import React, { memo, useState, useRef, useEffect } from 'react';
import style from './index.less';

// 触发onchange的条件:
// 1.键盘收起
// 2.输入框失去焦点
const Index = memo(({ placeholder, onChange, value, maxLength = 100, rows = 2 }) => {
  const [text, setText] = useState('');
  const inputRef = useRef(null);
  const inputWrapRef = useRef(null);

  useEffect(() => {
    inputWrapRef.current.bodyHeight = document.body.getBoundingClientRect().height;
  }, []);
  useEffect(() => {
    const onResize = () => {
      const currentBodyHeight = document.body.getBoundingClientRect().height;
      if (currentBodyHeight < inputWrapRef.current.bodyHeight) {
        console.log('键盘弹出了。。。。');
      } else if (inputWrapRef.current.style.visibility === 'visible') {
        // 键盘收起，并且输入框还是可见的情况下，要隐藏输入框
        console.log('键盘收起了。。。。');
        inputWrapRef.current.style.visibility = 'hidden';
        inputRef.current.blur();
        onChange(text);
      }
    };
    window.addEventListener('resize', onResize);
    return () => window.removeEventListener('resize', onResize);
  }, [onChange, text]);

  useEffect(() => {
    if (value !== text) {
      setText(value);
    }
  }, [value]);

  return (
    <React.Fragment>
      <div
        className={cs('row')}
        onClick={() => {
          inputWrapRef.current.style.visibility = 'visible';
          inputRef.current.focus();
        }}
      >
        { value || placeholder}
      </div>
      <div ref={inputWrapRef} className={cs('input-wrap')}>
        <textarea
          ref={inputRef}
          maxLength={maxLength}
          rows={rows}
          placeholder={placeholder || '请输入备注'}
          className={cs('input')}
          onChange={e => setText(e.target.value)}
          onBlur={() => {
            inputWrapRef.current.style.visibility = 'hidden';
            onChange(text);
          }}
        >
          {text}
        </textarea>
        <div>
          这里可以自定义内容，这部分内容也会被键盘撑起来。。
        </div>
      </div>
    </React.Fragment>
  );
});
export default Index;

```

使用：
```jsx
import React, { useEffect, useState, useCallback } from 'react';
import Input from './input';

const View = React.memo((props) => {
  const [inputText, setInputText] = useState('');
  const onInputChange = useCallback((value) => { setInputText(value); }, []);
  return (
    <div
      className={cs('container')}
    >
        <Input
          placeholder="占位的"
          value={inputText}
          onChange={onInputChange}
        />
    </div>
  );
});
export default View;
```
